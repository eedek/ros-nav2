Install navigation stack tools 
sudo apt install ros-jazzy-navigation2 ros-jazzy-nav2-bringup ros-jazzy-turtlebot3*

Add model export in bashrc
export TURTLEBOT3_MODEL=waffle
printenv | grep TURTLE

running simulation
ros2 launch turtlebot3_gazebo turtlebot3_world.launch.py

running teloeop teleop_keyboard
ros2 run turtlebot3_teleop teleop_keyboard

kill gazebo proceses if gazebo is not working
pkill -9 ruby && pkill -9 gz

run nan2 mapping
ros2 launch turtlebot3_cartographer cartographer.launch.py use_sim_time:=True

save generated map
ros2 run nav2_map_server map_saver_cli -f maps/my_map

ros2 launch turtlebot3_navigation2 navigation2.launch.py use_sim_time:=True map:=my_map.yaml
